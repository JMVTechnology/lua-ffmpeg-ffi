ffmpeg-ffi

LuaJIT FFI bindings to FFmpeg libraries.

Usage

local ffmpeg = require('ffmpeg')

ffmpeg.new('./example.mpg')
  :filter('gray', 'scale=40:12')
  :read_video_frame()
  :to_ascii()
  :and_then(function(ascii_frame)
    print(ascii_frame)
  end)
  :catch(function(err)
    print('An error occurred: ' .. err)
  end)
